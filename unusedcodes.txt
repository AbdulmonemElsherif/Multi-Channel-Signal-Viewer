function plotMainSignal(data, graphElement) {
    let i = 0;
    const interval = setInterval(() => {
      if (i < data.length) {
        const row = data[i];
        Plotly.extendTraces(graphElement, { x: [[row[0]]], y: [[row[1]]] }, [0]);
        i++;
      } else {
        clearInterval(interval);
      }
    }, 100);
}
